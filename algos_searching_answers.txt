function binarySearch(array, word) {
  let left = 0
  let right = array.length - 1
  if (word < array[left] || word > array[right]) {
    return false
  }
  while (left <= right) {
    let middle = Math.floor ((left + right)/2)
    if (array[middle] === word) {
      return middle
    } else if (array[middle] < word) {
      left = middle + 1
    } else {
      right = middle - 1
    }
  }
  return 'This word could not be found';
}

binarySearch(['aqua','albatross','cat','female'], 'albatross')

function searchFish(array, target) {
  for (let i = 0; i < array.length; i++) {
    if(array[i] === target) {
      return "Congrats! You got the white-spotted fish!"
    }
  }
  return "Nice try"
}

searchFish(['goldfish','clownfish','white spotted fish','tuna'], 'white spotted fish')
